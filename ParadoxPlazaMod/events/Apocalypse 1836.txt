#Outbreak
country_event = {
	id = 9999001
	title = "OUTBREAK_START_TITLE"
	desc = "OUTBREAK_START_DESC"
	picture = disease

	fire_only_once = yes
	
	trigger = {
		ai = no
		NOT = {
			has_country_flag = zombie_outbreak
			has_country_flag = zombie_outbreak_defeat
			has_country_flag = zombie_outbreak_victory
		}
		has_global_flag = zombie_mod
	}

	mean_time_to_happen = {
		days = 1
	}

	option = {
		name = "OUTBREAK_START_OPT"
		random_country = {
			limit = {
				NOT = { tag = SOK }
				number_of_states = 3
			}
			country_event = 9999002
		}
	}
}

#Country to Province
country_event = {
	id = 9999002
	title = "OUTBREAK_START_LOCAL_TITLE"
	desc = "OUTBREAK_START_LOCAL_DESC"
	picture = disease

	trigger = {
		any_neighbor_country = {
			has_country_flag = zombie_outbreak
		}
		NOT = {
			has_country_flag = zombie_outbreak
			has_country_flag = zombie_outbreak_defeat
			has_country_flag = zombie_outbreak_victory
			any_owned_province = {
				has_province_modifier = zombie_outbreak
			}
		}
		has_global_flag = zombie_mod
	}

	mean_time_to_happen = {
		months = 2
	}

	option = {
		name = "OUTBREAK_START_LOCAL_OPT"
		set_country_flag = zombie_outbreak
		any_pop = {
			consciousness = 1
			militancy = -2
		}
		random_province = {
			limit = {
				is_capital = no
				is_colonial = no
				is_overseas = no
			}
			any_pop = {
				reduce_pop = 0.8
				consciousness = -3
				militancy = 10
			}
			add_province_modifier = {
				name = zombie_outbreak
				duration = -1
			}
		}
	}
}

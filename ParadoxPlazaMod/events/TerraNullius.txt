province_event = {
	id = 99100
	title = "The Black Hole of $PROVINCENAME$"
	desc = "$PROVINCENAME$ has recently fallen under zombie control, and neither the local government nor the military has been able to restore order. Since we are unable to maintain control over the area, our people and the international community deem it terra nullius. Future rehabitation will require extensive reconstruction and repopulation; we estimate that the province's human population is less than 20% of what it was in 1836, if not lower."

	trigger = {
		province_control_days=1800 #5 years
		NOT = { owner = { tag = ZOM } }
		has_province_modifier = zombie_outbreak
		controlled_by = REB
		exists = ZOM
		ZOM = {
			has_country_modifier = accepts_new_provinces
		}
		NOT = { has_global_flag = no_zombie_mod }
	}

	mean_time_to_happen = {
		months = 24
		modifier = {
			factor = 0.75
			any_neighbor_province = {
				has_province_modifier = zombie_outbreak
			}
		}
		modifier = {
			factor = 0.75
			has_empty_adjacent_province = yes
		}
	}

	option = {
		name = "Uh oh."
		any_pop = {
			reduce_pop = 0.4
			militancy = 8
			consciousness = -3
		}
		add_core = THIS
		owner = { prestige = -3 }
		life_rating = -10
		change_controller = ZOM
		secede_province = ZOM
		ZOM = {
			remove_country_modifier = do_not_annex
			add_country_modifier = {
				name = do_not_annex
				duration = 31
			}
		}
	}
}

country_event = { #note, this must be a country event, otherwise it might happen multiple times at once, which can cause a crash
	id = 99101
	title = "Zombieland"
	desc = "Parts of our country have recently fallen under zombie control, and neither the local government nor the military have been able to restore order. Since we are unable to maintain control over the area, our people and the international community have come to refer to it as 'Zombieland': A wasteland where all societal structure has collapsed and where zombies roam freely. Future rehabitation will require extensive reconstruction and repopulation; we estimate that the province's human population is less than 20% of what it was in 1836, if not lower."

	trigger = {
		NOT = { exists = ZOM }
		any_owned_province = {
			province_control_days=1800 #5 years
			has_province_modifier = zombie_outbreak
			controlled_by = REB
		}
		NOT = { has_global_flag = no_zombie_mod }
	}

	mean_time_to_happen = {
		months = 24
	}

	option = {
		name = "Uh oh."
		random_owned = {
			limit = {
				province_control_days=1800 #5 years
				has_province_modifier = zombie_outbreak
				controlled_by = REB
			}
			any_pop = {
				reduce_pop = 0.4
				militancy = 8
				consciousness = -3
			}
			add_core = THIS
			owner = { prestige = -3 }
			life_rating = -10
			change_controller = ZOM
			secede_province = ZOM
		}
		ZOM = {
			remove_country_modifier = do_not_annex
			remove_country_modifier = accepts_new_provinces
			add_country_modifier = {
				name = do_not_annex
				duration = 62
			}
			country_event = { id = 99202 days = 31 }
		}
	}
}
